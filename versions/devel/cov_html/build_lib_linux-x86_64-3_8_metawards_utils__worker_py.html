19def prepare_worker(params: Parameters, demographics: Demographics, 

20 options: _Dict[str, any]) -> _Union[Network, Networks]: 

21 """Prepare a worker to receive work to run a model using the passed 

22 parameters. This will build the network specified by the 

23 parameters and will store it in global memory ready to 

24 be used for a model run. Note that these are 

25 silent, printing nothing to stdout or stderr 

26 

27 Parameters 

28 ---------- 

29 params: Parameters 

30 Parameters used to build the network 

31 demographics: Demographics 

32 If not None, then demographics used to specialise the Network 

33 into Networks 

34 """ 

35 # switch off printing to stdout and stderr 

36 global global_network 

37 

38 max_nodes = options["max_nodes"] 

39 max_links = options["max_links"] 

40 nthreads = options["nthreads"] 

41 

42 del options["max_nodes"] 

43 del options["max_links"] 

44 

45 profiler = options["profiler"] 

46 

47 if global_network is None: 

48 network = Network.build(params=params, 

49 profiler=profiler, 

50 max_nodes=max_nodes, 

51 max_links=max_links) 

52 

53 if demographics is not None: 

54 network = demographics.specialise(network, 

55 nthreads=nthreads, 

56 profiler=profiler) 

57 

58 global_network = network 

59 

60 # always work in a copy 

61 network = global_network.copy() 

62 network.update(params=params, demographics=demographics, 

63 nthreads=nthreads, profiler=profiler) 

64 

65 return network

68def run_worker(arguments): 

69 """Ask the worker to run a model using the passed variables and 

70 options. This will write to options['output_dir'] and will 

71 also return the population object that contains the final 

72 population data. 

73 

74 WARNING - the iterator and extractor arguments rely on the 

75 workers starting in the same directory as the main process, 

76 so that they can load the same python files (if the user 

77 is using a custom iterator or extractor) 

78 """ 

79 params = arguments["params"] 

80 demographics = arguments["demographics"] 

81 options = arguments["options"]

91 prompt=None, auto_bzip=auto_bzip) as output_dir: 

92 with Console.redirect_output(outdir=outdir, auto_bzip=auto_bzip): 

93 try: 

94 # first, build and prepare the Network(s). This is built once 

95 # from the parameters and demographics by loading files from 

96 # the filesystem, as sending this over the physical network 

97 # would be too expensive. Subsequent calls to this function 

98 # after the Network(s) has been built will call
